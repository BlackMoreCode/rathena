	mes "[CopeRO Headgear Crafter]";
	mes "Craft exclusive headgear using your coins and materials!";
	next;

	switch( select("Craft CopeRO headgears", "Leave") ) {
	case 1:
		mes "[CopeRO Headgear Crafter]";
		mes "Select the headgear you want to craft:";
		close2;
		callshop "barter_copero_headgear";
		end;
	case 2:
		mes "[CopeRO mid game gear shop]";
		mes "Select what you need / 필요한 아이템을 고르세요:";
		close2;
		callshop "barter_copero_weapon_gears";
		end;
	case 3:
		close;
	}
	end;
OnInit:
	setunittitle getnpcid(0), "<Headgear Crafter>";
	end;
}
